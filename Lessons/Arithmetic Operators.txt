 Arithmetic Operators = Operands (values, variables, etc.)
                        Operators (+ - * /)
                        ex. 11 = x + 5

 let students = 30;

 students = students + 1;      addition
 students = students - 1;      subtraction
 students = students * 2;      multiplication
 students = students / 2;      division
 students = students ** 2;     exponents
 students = students % 2;      remainder/modulus

 Augmented Assignment Operators = shorthand notations that combine a binary operation and an assignment into a single operator.

 students += 1;
 students -= 1;
 students *= 2;
 students /= 2 ;
 students **= 2;
 students %= 2;

 students++;   increment operator
 students--;   decrement operator


    operator precedence
    1. parenthesis ()
    2. exponents
    3. multiplication & division & modulo
    4. addition & subtraction
